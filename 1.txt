-- Create Tables
CREATE TABLE Doctors (
    DoctorID SERIAL PRIMARY KEY,
    Name VARCHAR(100),
    Specialization VARCHAR(100),
    Experience INT
);

CREATE TABLE Patients (
    PatientID SERIAL PRIMARY KEY,
    Name VARCHAR(100),
    Age INT,
    Gender VARCHAR(10),
    AdmissionDate DATE,
    DoctorID INT,
    FOREIGN KEY (DoctorID) REFERENCES Doctors(DoctorID)
);

-- Fetch Patient Details
SELECT * FROM Patients;

-- List Patients Admitted on a Specific Date ('2024-06-15')
SELECT * 
FROM Patients
WHERE AdmissionDate = '2024-06-15';

-- List Patients by Age Group
SELECT Name, Age,
    CASE
        WHEN Age < 18 THEN 'Child'
        WHEN Age BETWEEN 18 AND 60 THEN 'Adult'
        ELSE 'Senior'
    END AS AgeGroup
FROM Patients;

-- Update Patient Details (Change Name for PatientID = 1)
UPDATE Patients
SET Name = 'Alice Updated'
WHERE PatientID = 1;

-- List Doctors by Specialization ('Cardiology')
SELECT * 
FROM Doctors
WHERE Specialization = 'Cardiology';

-- Count Patients per Doctor
SELECT 
    d.Name AS DoctorName,
    COUNT(p.PatientID) AS PatientCount
FROM Doctors d
LEFT JOIN Patients p ON d.DoctorID = p.DoctorID
GROUP BY d.Name;

-- Calculate Average Experience of Doctors
SELECT AVG(Experience) AS AverageExperience
FROM Doctors;

-- Find Doctors with More Than 1 Patient
SELECT 
    d.Name,
    COUNT(p.PatientID) AS PatientCount
FROM Doctors d
JOIN Patients p ON d.DoctorID = p.DoctorID
GROUP BY d.Name
HAVING COUNT(p.PatientID) > 1;

-- Oldest Patient in the Hospital
SELECT *
FROM Patients
ORDER BY Age DESC
LIMIT 1;

-- Rank Doctors by Experience
SELECT 
    Name,
    Specialization,
    Experience,
    RANK() OVER (ORDER BY Experience DESC) AS Rank
FROM Doctors;
